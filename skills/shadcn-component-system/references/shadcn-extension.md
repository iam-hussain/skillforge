# Extending Shadcn UI Components

Patterns for wrapping, extending, and composing Shadcn UI primitives
without forking the source files.

---

## The Extension Rule

Never modify files in `components/ui/` that were generated by
`npx shadcn@latest add`. Instead, create wrapper components that extend
the originals using CVA composition and prop forwarding.

**Why:** Shadcn components are meant to be owned by your project, but
modifying generated files creates merge pain when updating. Wrapping
preserves the upgrade path.

---

## Pattern 1: Variant Extension

Add new variants to a Shadcn component by wrapping its existing CVA
schema.

```tsx
// components/ui/Button.tsx — generated by Shadcn (don't modify)
// Already exports: Button, buttonVariants

// components/shared/IconButton.tsx — your extension
import * as React from "react"
import { cva, type VariantProps } from "class-variance-authority"
import { cn } from "@/lib/utils"
import { buttonVariants } from "@/components/ui/Button"

const iconButtonVariants = cva(
  // Start from Shadcn's ghost+icon as base
  buttonVariants({ variant: "ghost", size: "icon" }),
  {
    variants: {
      color: {
        default: "text-muted-foreground hover:text-foreground",
        primary: "text-primary hover:text-primary/80",
        destructive: "text-destructive hover:text-destructive/80",
      },
      rounded: {
        true: "rounded-full",
        false: "",
      },
    },
    defaultVariants: {
      color: "default",
      rounded: false,
    },
  }
)

interface IconButtonProps
  extends React.ComponentPropsWithoutRef<"button">,
    VariantProps<typeof iconButtonVariants> {
  label: string // accessibility — always required
}

const IconButton = React.forwardRef<HTMLButtonElement, IconButtonProps>(
  ({ color, rounded, label, className, children, ...props }, ref) => {
    return (
      <button
        ref={ref}
        aria-label={label}
        className={cn(iconButtonVariants({ color, rounded }), className)}
        {...props}
      >
        {children}
      </button>
    )
  }
)
IconButton.displayName = "IconButton"

export { IconButton, iconButtonVariants }
export type { IconButtonProps }
```

**Usage:**
```tsx
import { Trash2, Settings } from "lucide-react"
import { IconButton } from "@/components/shared/IconButton"

<IconButton label="Delete item" color="destructive">
  <Trash2 className="h-4 w-4" />
</IconButton>

<IconButton label="Settings" rounded>
  <Settings className="h-4 w-4" />
</IconButton>
```

## Pattern 2: Composition Wrapper

Combine multiple Shadcn primitives into a higher-level molecule.

```tsx
// components/shared/ConfirmDialog.tsx
import * as React from "react"
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/AlertDialog"
import { buttonVariants } from "@/components/ui/Button"
import { cn } from "@/lib/utils"

interface ConfirmDialogProps {
  trigger: React.ReactNode
  title: string
  description: string
  confirmLabel?: string
  cancelLabel?: string
  variant?: "default" | "destructive"
  onConfirm: () => void | Promise<void>
}

export function ConfirmDialog({
  trigger,
  title,
  description,
  confirmLabel = "Confirm",
  cancelLabel = "Cancel",
  variant = "default",
  onConfirm,
}: ConfirmDialogProps) {
  return (
    <AlertDialog>
      <AlertDialogTrigger asChild>{trigger}</AlertDialogTrigger>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>{title}</AlertDialogTitle>
          <AlertDialogDescription>{description}</AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter>
          <AlertDialogCancel>{cancelLabel}</AlertDialogCancel>
          <AlertDialogAction
            className={cn(buttonVariants({ variant }))}
            onClick={onConfirm}
          >
            {confirmLabel}
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  )
}
```

**Usage:**
```tsx
<ConfirmDialog
  trigger={<Button variant="destructive">Delete Account</Button>}
  title="Are you sure?"
  description="This action cannot be undone."
  confirmLabel="Delete"
  variant="destructive"
  onConfirm={handleDelete}
/>
```

## Pattern 3: Enhanced Data Display

Extend Shadcn's `Table` with sorting, empty states, and loading
skeleton — all as variant-driven props.

```tsx
// components/shared/DataTable.tsx
import * as React from "react"
import { cva, type VariantProps } from "class-variance-authority"
import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/Table"
import { Skeleton } from "@/components/ui/Skeleton"
import { cn } from "@/lib/utils"

const dataTableVariants = cva("w-full", {
  variants: {
    density: {
      compact: "[&_td]:py-1.5 [&_th]:py-1.5",
      default: "[&_td]:py-3 [&_th]:py-3",
      relaxed: "[&_td]:py-4 [&_th]:py-4",
    },
    striped: {
      true: "[&_tr:nth-child(even)]:bg-muted/50",
      false: "",
    },
  },
  defaultVariants: {
    density: "default",
    striped: false,
  },
})

interface Column<T> {
  key: keyof T & string
  header: string
  render?: (value: T[keyof T], row: T) => React.ReactNode
  className?: string
}

interface DataTableProps<T>
  extends VariantProps<typeof dataTableVariants> {
  columns: Column<T>[]
  data: T[]
  loading?: boolean
  loadingRows?: number
  emptyMessage?: string
  className?: string
}

export function DataTable<T extends Record<string, unknown>>({
  columns,
  data,
  loading = false,
  loadingRows = 5,
  emptyMessage = "No results found.",
  density,
  striped,
  className,
}: DataTableProps<T>) {
  return (
    <div className={cn(dataTableVariants({ density, striped }), className)}>
      <Table>
        <TableHeader>
          <TableRow>
            {columns.map((col) => (
              <TableHead key={col.key} className={col.className}>
                {col.header}
              </TableHead>
            ))}
          </TableRow>
        </TableHeader>
        <TableBody>
          {loading
            ? Array.from({ length: loadingRows }).map((_, i) => (
                <TableRow key={i}>
                  {columns.map((col) => (
                    <TableCell key={col.key}>
                      <Skeleton className="h-4 w-full" />
                    </TableCell>
                  ))}
                </TableRow>
              ))
            : data.length === 0
              ? (
                  <TableRow>
                    <TableCell
                      colSpan={columns.length}
                      className="h-24 text-center text-muted-foreground"
                    >
                      {emptyMessage}
                    </TableCell>
                  </TableRow>
                )
              : data.map((row, i) => (
                  <TableRow key={i}>
                    {columns.map((col) => (
                      <TableCell key={col.key} className={col.className}>
                        {col.render
                          ? col.render(row[col.key], row)
                          : String(row[col.key] ?? "")}
                      </TableCell>
                    ))}
                  </TableRow>
                ))}
        </TableBody>
      </Table>
    </div>
  )
}
```

**Usage:**
```tsx
<DataTable
  columns={[
    { key: "name", header: "Name" },
    { key: "status", header: "Status", render: (_, row) => <StatusBadge status={row.status} dot /> },
    { key: "amount", header: "Amount", className: "text-right" },
  ]}
  data={orders}
  loading={isLoading}
  density="compact"
  striped
/>
```

## Pattern 4: Form Field Wrapper

Combine Shadcn's `Label` and form primitives with validation state
driven by CVA.

```tsx
// components/shared/FormField.tsx
import * as React from "react"
import { cva, type VariantProps } from "class-variance-authority"
import { Label } from "@/components/ui/Label"
import { cn } from "@/lib/utils"

const formFieldVariants = cva("space-y-1.5", {
  variants: {
    state: {
      default: "",
      error: "[&_input]:border-destructive [&_textarea]:border-destructive",
      // Note: emerald is used because Shadcn has no --success token by default.
      // Add `--success: 160 84% 39%;` to globals.css to fully tokenize.
      success: "[&_input]:border-emerald-500 [&_textarea]:border-emerald-500",
    },
  },
  defaultVariants: { state: "default" },
})

interface FormFieldProps
  extends VariantProps<typeof formFieldVariants> {
  label: string
  error?: string
  hint?: string
  required?: boolean
  children: React.ReactNode
  className?: string
}

export function FormField({
  label,
  error,
  hint,
  required,
  state,
  children,
  className,
}: FormFieldProps) {
  const id = React.useId()
  const resolvedState = error ? "error" : state

  return (
    <div className={cn(formFieldVariants({ state: resolvedState }), className)}>
      <Label htmlFor={id}>
        {label}
        {required && <span className="ml-0.5 text-destructive">*</span>}
      </Label>
      {React.cloneElement(children as React.ReactElement, { id })}
      {error && <p className="text-sm text-destructive">{error}</p>}
      {!error && hint && <p className="text-sm text-muted-foreground">{hint}</p>}
    </div>
  )
}
```

## Extension Decision Guide

| Need | Pattern | Location |
|------|---------|----------|
| New visual variants for existing component | Variant Extension | `components/shared/` |
| Combine 2+ Shadcn primitives | Composition Wrapper | `components/shared/` |
| Add data handling (loading, empty, sort) | Enhanced Data Display | `components/shared/` |
| Wrap with form validation state | Form Field Wrapper | `components/shared/` |
| Feature-specific composition | Organism-level wrapper | `features/*/components/` |

---

## Rules

1. **Never modify generated Shadcn files** — always wrap
2. **Export the wrapper's CVA schema** — consumers may need it
3. **Preserve native HTML attrs** — use `ComponentPropsWithoutRef`
4. **Keep wrappers under 100 lines** — if bigger, decompose further
5. **One wrapper per file** — co-locate its CVA schema in the same file
6. **Named exports only** — no default exports
